### 크롤링 예제 (Graal JS)

```javascript
/**
 * @typedef {object} LLMScore
 * @property {number} rank
 * @property {string} model
 * @property {number} score
 */


const BASE_URL = "https://lmarena.ai/leaderboard/";
const TYPES = {
    "text": "텍스트 기반 응답 선호도 순위",
    "webdev": "웹 개발 작업 선호도 순위",
    "vision": "시각 처리 능력 선호도 순위",
    "text-to-image": "텍스트 프롬프트 기반 이미지 생성 선호도 순위",
    "image-edit": "제공된 이미지 기반 이미지 생성 선호도 순위",
    "search": "인터넷 검색에 기반한 응답 선호도 순위",
    "text-to-video": "텍스트 프롬프트 기반 동영상 생성 선호도 순위",
    "image-to-video": "제공된 이미지 기반 동영상 생성 선호도 순위"
};



/**
 * @description LM Arena에서 분야별 점수 조회
 * @param {string} [type="text"] TYPES 참고
 * @returns {LLMScore[]|null}
 */
function getArenaRank(type) {
    type = type || "text";
    if (!TYPES[type]) return null;

    const url = BASE_URL + type;
    let html;
    try {
        html = org.jsoup.Jsoup.connect(url).get();
    } catch (e) {
        Log.e(`${e.name}\n${e.message}\n${e.stack}`);
        return null;
    }

    const result = html.select("tbody > tr")
        .toArray()
        .map(tr => {
            const td = tr.select("td");
            const rank = Number(td.get(0).text());
            const model = String(td.get(2).text()); // 그냥 쓰면 Java string이라 귀찮아질 수 있음
            const score = Number(td.get(3).text());
            return { rank, model, score };
        });

    return result;
}
```
